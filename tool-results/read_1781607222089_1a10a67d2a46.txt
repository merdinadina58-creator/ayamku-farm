   352→  // Computed values for selected batch
   353→  const getBatchStats = (batch: Batch) => {
   354→    const totalFeed = batch.feedRecords.reduce((s, f) => s + f.quantityKg, 0)
   355→    const totalCost = batch.feedRecords.reduce((s, f) => s + f.quantityKg * f.pricePerKg, 0)
   356→    const totalDead = batch.mortalityRecords.reduce((s, m) => s + m.quantity, 0)
   357→    const aliveCount = batch.quantity - totalDead
   358→    const latestWeight = batch.weightRecords.length > 0
   359→      ? batch.weightRecords.reduce((latest, r) => new Date(r.date) > new Date(latest.date) ? r : latest, batch.weightRecords[0]).averageWeightGram
   360→      : batch.initialWeight * 1000
   361→    const weightGain = latestWeight - batch.initialWeight * 1000
   362→    const fcr = weightGain > 0 && aliveCount > 0 ? (totalFeed * 1000) / (aliveCount * weightGain) : 0
   363→    const now = new Date()
   364→    const arrival = new Date(batch.arrivalDate)
   365→    const ageDays = Math.floor((now.getTime() - arrival.getTime()) / (1000 * 60 * 60 * 24))
   366→    const feedPerEkor = aliveCount > 0 ? totalFeed / aliveCount : 0
   367→    const mortalityRate = batch.quantity > 0 ? (totalDead / batch.quantity) * 100 : 0

   373→    const totalHarvestKg = harvestQty * harvestWt
   374→    const totalHarvestValue = totalHarvestKg * sellPrice
   375→    const profit = totalHarvestValue - totalCost
   376→

   378→      totalFeed, totalCost, latestWeight, weightGain, fcr, ageDays,
   379→      feedPerEkor, totalDead, aliveCount, mortalityRate,
   380→      harvestQty, harvestWt, sellPrice, totalHarvestKg, totalHarvestValue, profit,
   381→    }
