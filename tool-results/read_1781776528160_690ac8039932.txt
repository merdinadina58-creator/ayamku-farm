   541→  const handleLogoUpload = (e: React.ChangeEvent<HTMLInputElement>) => {
   542→    const file = e.target.files?.[0]

   545→      toast({ title: 'Error', description: 'File harus berupa gambar', variant: 'destructive' })
   546→      return
   547→    }
