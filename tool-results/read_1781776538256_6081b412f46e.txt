   598→   598→    const totalFeed = batch.feedRecords.reduce((s, f) => s + f.quantityKg, 0)
   599→   599→    const totalCost = batch.feedRecords.reduce((s, f) => s + f.quantityKg * f.pricePerKg, 0)
   600→   600→    const totalDead = batch.mortalityRecords.reduce((s, m) => s + m.quantity, 0)
   601→   601→    const aliveCount = batch.quantity - totalDead
   602→   602→    const latestWeight = batch.weightRecords.length > 0
   603→   603→      ? batch.weightRecords.reduce((latest, r) => new Date(r.date) > new Date(latest.date) ? r : latest, batch.weightRecords[0]).averageWeightGram
   604→   604→      : batch.initialWeight * 1000
   605→   605→    const weightGain = latestWeight - batch.initialWeight * 1000
   606→   606→    const fcr = weightGain > 0 && aliveCount > 0 ? (totalFeed * 1000) / (aliveCount * weightGain) : 0
   607→   607→    const now = batch.status === 'harvested' && batch.harvestDate ? new Date(batch.harvestDate) : new Date()
   608→   608→    const arrival = new Date(batch.arrivalDate)
   609→   609→    const ageDays = Math.floor((now.getTime() - arrival.getTime()) / (1000 * 60 * 60 * 24))
   610→   610→    const feedPerEkor = aliveCount > 0 ? totalFeed / aliveCount : 0

   616→   616→    const sellPrice = batch.sellingPricePerKg || 0
   617→   617→    const totalHarvestKg = harvestQty * harvestWt

   619→   619→    const profit = totalHarvestValue - totalCost
   620→   620→
   621→   621→    return {
   622→   622→      totalFeed, totalCost, latestWeight, weightGain, fcr, ageDays,
   623→   623→      feedPerEkor, totalDead, aliveCount, mortalityRate,
   624→   624→      harvestQty, harvestWt, sellPrice, totalHarvestKg, totalHarvestValue, profit,
